package Beans;

public class ElementoClase extends ClaseEnemigos{

    private int idElemento;
    private String nombreElemento;
    private float daño;
    private String fd;

    public int getIdClase() {return idClase}

    public void setIdElemento(int idElemento) {this.idElemento = idElemento}

    public String getNombreElemento() {
        return nombreElemento
    }

    public void setNombreElemento(String nombreElemento) {
        this.nombreElemento = nombreElemento
    }

    public float getDaño() {return daño}

    public void setDaño(int daño) {this.daño = daño}

    public String getFd() {
        return fd
    }

    public void setFd(String fd) {
        this.fd = fd
    }
}
}
